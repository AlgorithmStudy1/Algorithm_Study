cnt =1
while 1:
  L, P, V = map(int, input().split())
  if L == 0 and P == 0 and V == 0:
    break
  res = (V//P)*L
  res += min((V % P), L)
  print('Case %d: %d' % (cnt, res))
  cnt += 1
